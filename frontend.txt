The Updated useCartStore.ts
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';

interface CartItem {
  skuId: string;
  partName: string;
  quantity: number;
  wholesalePrice: number; // in cents
}

interface CartState {
  guestId: string | null;
  items: CartItem[];
  addItem: (item: CartItem) => void;
  removeItem: (skuId: string) => void;
  clearCart: () => void;
  initializeGuest: () => void;
}

export const useCartStore = create<CartState>()(
  persist(
    (set, get) => ({
      guestId: null,
      items: [],

      initializeGuest: () => {
        if (!get().guestId) {
          // Generates a 5-digit padded sequence based on timestamp for the demo
          const sequence = Math.floor(Math.random() * 90000) + 10000;
          const newId = `userid-g${sequence}`;
          set({ guestId: newId });
        }
      },

      addItem: (newItem) => {
        get().initializeGuest(); // Ensure ID exists on first interaction
        const currentItems = get().items;
        const existingItem = currentItems.find((i) => i.skuId === newItem.skuId);

        if (existingItem) {
          set({
            items: currentItems.map((i) =>
              i.skuId === newItem.skuId
                ? { ...i, quantity: i.quantity + newItem.quantity }
                : i
            ),
          });
        } else {
          // Enforce 5-unit wholesale MOQ on initial add
          const quantity = newItem.quantity >= 5 ? newItem.quantity : 5;
          set({ items: [...currentItems, { ...newItem, quantity }] });
        }
      },

      removeItem: (skuId) => 
        set({ items: get().items.filter((i) => i.skuId !== skuId) }),

      clearCart: () => set({ items: [] }),
    }),
    {
      name: 'ct-cart-storage',
      storage: createJSONStorage(() => localStorage),
    }
  )
);
